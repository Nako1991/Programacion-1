//TEST DE LIBRERIA CONIO
void dibujarCuadro(int,int,int,int);

void dibujarCuadro(int x1,int x2,int y1,int y2)
{
    int i;
    for (i=x1+1; i<x2; i++)
    {
        gotoxy(i,y1);
        cprintf("-") ;//linea horizontal superior
        gotoxy(i,y2);
        cprintf("-") ;//linea horizontal inferior
    }

    for (i=y1+1; i<y2; i++)
    {
        gotoxy(x1,i);
        cprintf("|") ;//linea vertical izquierda
        gotoxy(x2,i);
        cprintf("|") ;//linea vertical derecha
    }
}

void dibujarLinea(int x1,int x2,int y1,int y2)
{
    int i;
    if ( y1 == y2 )
    {
        for ( i=x1 ; i<=x2 ; i++ )
        {
            gotoxy(i,y1);
            cprintf("-");//linea horizontal
        }
    }
    else if ( x1 == x2 )
    {
        for ( i=y1 ; i<=y2 ; i++ )
        {
            gotoxy(x1,i);
            cprintf("|");//linea vertical
        }
    }
}
